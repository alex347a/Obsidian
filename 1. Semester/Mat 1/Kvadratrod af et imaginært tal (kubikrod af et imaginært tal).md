\begin{align*}
\text{Trin 1: Skriv } -8i \text{ i polarform.} \\
r &= 8, \quad \theta = \frac{3\pi}{2}. \\
-8i &= 8e^{i\frac{3\pi}{2}}. \\
\text{Da vinklen er på den negative imaginære akse}
\\
\text{Trin 2: Find kubikrødderne.} \\
z^{1/3} &= r^{1/3} e^{i\left(\frac{\theta + 2k\pi}{3}\right)}, \quad k = 0, 1, 2. \\
r^{1/3} &= 8^{1/3} = 2. \\
z_k &= 2 e^{i\frac{\frac{3\pi}{2} + 2k\pi}{3}} \quad \text{for} \quad k = 0, 1, 2. \\
\\
\text{Trin 3: Beregn rødderne.} \\
\text{For } k = 0: \quad z_0 &= 2 e^{i\frac{\frac{3\pi}{2}}{3}} = 2 e^{i\frac{\pi}{2}} = 2(\cos(\frac{\pi}{2}) + i\sin(\frac{\pi}{2})) = 2i. \\
\text{For } k = 1: \quad z_1 &= 2 e^{i\frac{\frac{3\pi}{2} + 2\pi}{3}} = 2 e^{i\frac{7\pi}{6}} = 2\left(\cos\left(\frac{7\pi}{6}\right) + i\sin\left(\frac{7\pi}{6}\right)\right) = -\sqrt{3} - i. \\
\text{For } k = 2: \quad z_2 &= 2 e^{i\frac{\frac{3\pi}{2} + 4\pi}{3}} = 2 e^{i\frac{11\pi}{6}} = 2\left(\cos\left(\frac{11\pi}{6}\right) + i\sin\left(\frac{11\pi}{6}\right)\right) = \sqrt{3} - i. \\
\\
\text{Samlet: De tre kubikrødder af } -8i \text{ er:} \\
z_0 &= 2i, \quad z_1 = -\sqrt{3} - i, \quad z_2 = \sqrt{3} - i.
\end{align*}
