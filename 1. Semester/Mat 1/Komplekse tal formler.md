$$
z = a + i \cdot b \quad \text{(Kartesisk form)}
$$

$$
z = m \cdot e^{i\theta} \quad \text{(Polær form)}
$$

$$
e^{i\theta} = \cos(\theta) + i \cdot \sin(\theta)
$$

$$
m = \sqrt{a^2 + b^2}
$$

$$
\theta = \arctan\left( \frac{b}{a} \right) + n \cdot \pi \quad n \in \{0, 1\} \text{ afhængig af } a \text{ og } b
$$
